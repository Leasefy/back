       ```prisma
       /// Immutable snapshot of scoring features captured at prediction time.
       /// Critical for ML training: never use mutable Application data for training.
       model ApplicationFeatureSnapshot {
         id               String   @id @default(uuid()) @db.Uuid
         applicationId    String   @unique @map("application_id") @db.Uuid
         features         Json     // ScoringFeatures interface serialized at scoring time
         propertyRent     Int      @map("property_rent") // Monthly rent at scoring time
         algorithmVersion String   @default("2.1") @map("algorithm_version")
         createdAt        DateTime @default(now()) @map("created_at")

         application      Application @relation(fields: [applicationId], references: [id], onDelete: Cascade)

         @@index([createdAt])
         @@index([algorithmVersion])
         @@map("application_feature_snapshots")
       }
       ```

    3. Add PredictionLog model to prisma/schema.prisma:
       ```prisma

       /// Outcomes evolve: APPROVED_PENDING -> APPROVED_PAID_ON_TIME | APPROVED_LATE_PAYMENTS | APPROVED_DEFAULTED
       model PredictionLog {
         id               String    @id @default(uuid()) @db.Uuid
         applicationId    String    @unique @map("application_id") @db.Uuid
         predictedScore   Int       @map("predicted_score")  // 0-100, captured at scoring time
         predictedLevel   RiskLevel @map("predicted_level")  // A/B/C/D
         actualOutcome    String?   @map("actual_outcome")   // PredictionOutcome values
         outcomeUpdatedAt DateTime? @map("outcome_updated_at")
         leaseId          String?   @map("lease_id") @db.Uuid
         monthsTracked    Int?      @map("months_tracked")
         latePaymentCount Int?      @map("late_payment_count")
         defaulted        Boolean   @default(false)
         algorithmVersion String    @default("2.1") @map("algorithm_version")
         createdAt        DateTime  @default(now()) @map("created_at")

         application      Application @relation(fields: [applicationId], references: [id], onDelete: Cascade)

         @@index([actualOutcome])
         @@index([createdAt])
         @@index([algorithmVersion])
         @@map("prediction_logs")
       ```

       ```typescript
       // 6a. ML Persistence: Snapshot features and log prediction
       try {
         await this.mlPersistenceService.createSnapshot(
           applicationId,
           features,
           application.property.monthlyRent,
           '2.1',
         );
         await this.mlPersistenceService.createPredictionLog(

           result.total,
           result.level,
           '2.1',
         );

           `Failed to persist ML data for application ${applicationId}: ${
             mlError instanceof Error ? mlError.message : 'Unknown error'
           }`,
         );
       }
       ```
