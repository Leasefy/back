/// Type of coupon discount
/// PERCENTAGE: Discount % off subscription price (1-100)

2. In `prisma/schema.prisma`, add in the MODELS section (after TenantPreference model):

```prisma
/// Coupon code for subscription discounts
/// Admin-managed with validation rules and usage tracking
model Coupon {
  id              String     @id @default(uuid()) @db.Uuid
  code            String     @unique @db.VarChar(50)
  type            CouponType

  // Discount values (only one used based on type)
  percentageOff   Int?       @map("percentage_off")   // 1-100 for PERCENTAGE
  amountOff       Int?       @map("amount_off")       // COP for FIXED_AMOUNT
  freeMonths      Int?       @map("free_months")      // Months for FREE_MONTHS

  // Validity
  validFrom       DateTime   @map("valid_from")
  validUntil      DateTime   @map("valid_until")

  // Usage limits
  maxUses         Int?       @map("max_uses")          // null = unlimited
  currentUses     Int        @default(0) @map("current_uses")

  // Plan applicability (empty = all plans)
  applicablePlans String[]   @default([]) @map("applicable_plans")

  // Description for admin
  description     String?    @db.Text

  isActive        Boolean    @default(true) @map("is_active")
  createdAt       DateTime   @default(now()) @map("created_at")
  updatedAt       DateTime   @updatedAt @map("updated_at")

  // Relations
  usages          CouponUsage[]

  @@index([code])
  @@index([isActive])
  @@index([validUntil])
  @@map("coupons")
}

/// Tracks coupon usage per user to prevent duplicate redemptions
model CouponUsage {
  id             String   @id @default(uuid()) @db.Uuid
  couponId       String   @map("coupon_id") @db.Uuid
  userId         String   @map("user_id") @db.Uuid
  subscriptionId String?  @map("subscription_id") @db.Uuid
  usedAt         DateTime @default(now()) @map("used_at")

  // Relations
  coupon         Coupon   @relation(fields: [couponId], references: [id], onDelete: Cascade)
  user           User     @relation(fields: [userId], references: [id])

  @@unique([couponId, userId])
  @@index([userId])
  @@map("coupon_usages")
}
```

**Services:**

5. `src/coupons/coupons.service.ts` (CouponsService):
- Inject PrismaService
- `create(dto: CreateCouponDto)`: Normalize code to uppercase with `dto.code.toUpperCase().trim()`. Create coupon in database with `prisma.coupon.create()`. Convert validFrom/validUntil strings to Date objects.
- `findAll()`: Return all coupons ordered by createdAt desc, include `_count: { select: { usages: true } }` for usage stats.
- `findById(id: string)`: Find coupon by ID or throw NotFoundException.
- `update(id: string, dto: UpdateCouponDto)`: Update coupon fields. Only updates provided fields.
- `deactivate(id: string)`: Set isActive=false (soft delete).
- `recordUsage(couponId: string, userId: string, subscriptionId: string)`: Transaction that atomically increments currentUses and creates CouponUsage record.

6. `src/coupons/coupon-validation.service.ts` (CouponValidationService):
- Inject PrismaService
- `validateCoupon(code: string, userId: string, planId: string)`: Returns `{ valid: boolean; message: string; coupon?: Coupon; discountPreview?: { type, percentageOff?, amountOff?, freeMonths? } }`
  - Normalize code: `code.toUpperCase().trim()`
  - Find coupon by code (include usages), return invalid if not found ("Codigo de cupon no valido")
  - Check isActive, return invalid if false ("Este cupon esta desactivado")
  - Check validFrom <= now <= validUntil, return invalid if outside range ("Este cupon aun no es valido" / "Este cupon ha expirado")
  - Check maxUses: if not null and currentUses >= maxUses, return invalid ("Este cupon ha alcanzado su limite de uso")
  - Check user hasn't used: find usage with userId in usages array, return invalid if found ("Ya has usado este cupon")
  - Check plan applicability: if applicablePlans is non-empty, load plan config by planId, build key as `${plan.planType}_${plan.tier}`, check if key in applicablePlans, return invalid if not ("Este cupon no es aplicable al plan seleccionado")
  - Return valid with coupon object and discount preview

7. `src/coupons/coupon-application.service.ts` (CouponApplicationService):
- Pure service, no dependencies needed
- `applyDiscount(originalPrice: number, coupon: { type: CouponType; percentageOff?: number | null; amountOff?: number | null; freeMonths?: number | null })`: Returns `{ finalPrice: number; discountAmount: number; freeMonths?: number }`
  - PERCENTAGE: discountAmount = Math.floor(originalPrice * percentageOff / 100), finalPrice = originalPrice - discountAmount
  - FIXED_AMOUNT: discountAmount = Math.min(amountOff, originalPrice), finalPrice = originalPrice - discountAmount
  - FREE_MONTHS: finalPrice = 0, discountAmount = originalPrice, freeMonths = coupon.freeMonths
  - FULL_ACCESS: finalPrice = 0, discountAmount = originalPrice
  - Ensure finalPrice >= 0 always
