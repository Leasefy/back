  @Cron('0 14 * * *')  // 9 AM Colombia = 14:00 UTC
  async checkTrialExpiry(): Promise<void> {
    // 1. Find subscriptions where status=TRIAL AND trialEndsAt is tomorrow
    //    (to send 1-day-before notification)
    const tomorrow = new Date();
    tomorrow.setDate(tomorrow.getDate() + 1);
    const startOfTomorrow = new Date(tomorrow); startOfTomorrow.setHours(0,0,0,0);
    const endOfTomorrow = new Date(tomorrow); endOfTomorrow.setHours(23,59,59,999);

    const expiringTrials = await this.prisma.subscription.findMany({
      where: {
        status: 'TRIAL',
        trialEndsAt: { gte: startOfTomorrow, lte: endOfTomorrow },
      },
    });

    for (const sub of expiringTrials) {
      await this.notificationsService.send({
        userId: sub.userId,
        templateCode: 'TRIAL_EXPIRING',
        variables: { date: this.formatDate(sub.trialEndsAt!) },
        triggeredBy: `scheduler:trial:${sub.id}`,
      });
    }

    // 2. Find trials that have already expired (trialEndsAt < now, status still TRIAL)
    //    -> Downgrade to free
    const expiredTrials = await this.prisma.subscription.findMany({
      where: {
        status: 'TRIAL',
        trialEndsAt: { lt: new Date() },
      },
    });

    for (const sub of expiredTrials) {
      // Call SubscriptionsService.handleExpiredSubscriptions() or handle inline
      await this.subscriptionsService.handleSingleExpiry(sub.id);

      await this.notificationsService.send({
        userId: sub.userId,
        templateCode: 'TRIAL_EXPIRED',
        variables: {},
        triggeredBy: `scheduler:trial-expired:${sub.id}`,
      });
    }
  }
  ```

  async handleSingleExpiry(subscriptionId: string): Promise<void> {
    const sub = await this.prisma.subscription.findUnique({
      where: { id: subscriptionId },
      include: { plan: true },
    });
    if (!sub) return;

    // Mark subscription as EXPIRED
    await this.prisma.subscription.update({
      where: { id: subscriptionId },
      data: { status: 'EXPIRED' },
    });

    // Downgrade user to FREE
    await this.prisma.user.update({
      where: { id: sub.userId },
      data: { subscriptionPlan: 'FREE' },
    });

    // If landlord: hide excess properties
    if (sub.plan.planType === 'LANDLORD') {
      await this.hideExcessProperties(sub.userId);
    }
  }

  private async hideExcessProperties(userId: string): Promise<void> {
    // Get all published properties (AVAILABLE or PENDING)
    const publishedProperties = await this.prisma.property.findMany({
      where: {
        landlordId: userId,
        status: { in: ['AVAILABLE', 'PENDING'] },
      },
      orderBy: { createdAt: 'asc' },  // Keep oldest (first published)
    });

    if (publishedProperties.length <= 1) return;

    // Keep first one published, set rest to DRAFT
    const toHide = publishedProperties.slice(1).map(p => p.id);
    await this.prisma.property.updateMany({
      where: { id: { in: toHide } },
      data: { status: 'DRAFT' },
    });
  }
  ```
