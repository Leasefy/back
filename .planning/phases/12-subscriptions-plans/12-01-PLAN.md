  **3. Add 4 new models to schema.prisma:**

  ```prisma
  /// Configurable subscription plan definition
  /// Admin can modify prices. Different plans for TENANT vs LANDLORD.
  model SubscriptionPlanConfig {
    id          String        @id @default(uuid()) @db.Uuid
    planType    PlanType      @map("plan_type")        // TENANT or LANDLORD
    tier        SubscriptionPlan                        // FREE, PRO, BUSINESS
    name        String        @db.VarChar(100)          // Display name: "Tenant Pro", "Landlord Business"
    description String?       @db.Text

    // Pricing (COP - 0 for free plan)
    monthlyPrice Int          @default(0) @map("monthly_price")
    annualPrice  Int          @default(0) @map("annual_price")   // Full year price (discounted)

    // Limits
    maxProperties     Int     @default(1) @map("max_properties")      // -1 = unlimited
    maxScoringViews   Int     @default(1) @map("max_scoring_views")   // Per month, -1 = unlimited
    hasPremiumScoring Boolean @default(false) @map("has_premium_scoring")
    hasApiAccess      Boolean @default(false) @map("has_api_access")

    // Micropayment price (for tenant free extra scoring views)
    scoringViewPrice  Int     @default(0) @map("scoring_view_price")  // COP per extra view

    isActive    Boolean       @default(true) @map("is_active")
    createdAt   DateTime      @default(now()) @map("created_at")
    updatedAt   DateTime      @updatedAt @map("updated_at")

    // Relations
    subscriptions Subscription[]

    @@unique([planType, tier])  // One config per role+tier combo
    @@index([planType])
    @@index([isActive])
    @@map("subscription_plan_configs")
  }

  /// User's active subscription
  model Subscription {
    id        String             @id @default(uuid()) @db.Uuid
    userId    String             @map("user_id") @db.Uuid
    planId    String             @map("plan_id") @db.Uuid
    status    SubscriptionStatus @default(TRIAL)
    cycle     BillingCycle       @default(MONTHLY)

    // Period
    startDate   DateTime         @map("start_date")
    endDate     DateTime         @map("end_date")
    trialEndsAt DateTime?        @map("trial_ends_at")  // Set only during trial

    // Cancellation
    cancelledAt   DateTime?      @map("cancelled_at")
    cancelReason  String?        @map("cancel_reason") @db.VarChar(500)

    // Auto-renew
    autoRenew   Boolean          @default(true) @map("auto_renew")

    createdAt   DateTime         @default(now()) @map("created_at")
    updatedAt   DateTime         @updatedAt @map("updated_at")

    // Relations
    plan     SubscriptionPlanConfig @relation(fields: [planId], references: [id])
    payments SubscriptionPayment[]

    @@index([userId])
    @@index([status])
    @@index([endDate])
    @@map("subscriptions")
  }

  /// Payment record for a subscription (via PSE mock)
  model SubscriptionPayment {
    id              String   @id @default(uuid()) @db.Uuid
    subscriptionId  String   @map("subscription_id") @db.Uuid
    amount          Int      // COP
    cycle           BillingCycle

    // PSE mock reference
    pseTransactionId String? @map("pse_transaction_id") @db.VarChar(100)
    status           String  @default("PENDING") @db.VarChar(20)  // PENDING, SUCCESS, FAILED

    // Period covered
    periodStart DateTime    @map("period_start")
    periodEnd   DateTime    @map("period_end")

    createdAt   DateTime    @default(now()) @map("created_at")

    // Relations
    subscription Subscription @relation(fields: [subscriptionId], references: [id], onDelete: Cascade)

    @@index([subscriptionId])
    @@index([status])
    @@map("subscription_payments")
  }

  /// Track tenant scoring view usage per month
  /// Enforces monthly limits for free tier tenants
  model ScoringUsage {
    id        String   @id @default(uuid()) @db.Uuid
    userId    String   @map("user_id") @db.Uuid
    month     Int      // 1-12
    year      Int      // e.g., 2026
    viewCount Int      @default(0) @map("view_count")  // Number of scoring views used

    // Micropayment tracking
    paidViewCount Int  @default(0) @map("paid_view_count")  // Extra views purchased via micropayment

    createdAt DateTime @default(now()) @map("created_at")
    updatedAt DateTime @updatedAt @map("updated_at")

    @@unique([userId, month, year])
    @@index([userId])
    @@map("scoring_usage")
  }
  ```

  **Default plans to seed:**

  | PlanType | Tier | Name | Monthly Price | Annual Price | Max Props | Max Scoring/mo | Premium Scoring | API | Scoring View Price |
  |----------|------|------|---------------|--------------|-----------|----------------|-----------------|-----|--------------------|
  | TENANT | FREE | Tenant Gratis | 0 | 0 | -1 (N/A) | 1 | false | false | 9900 |
  | TENANT | PRO | Tenant Pro | 49900 | 479000 | -1 (N/A) | -1 | true | false | 0 |
  | LANDLORD | FREE | Landlord Gratis | 0 | 0 | 1 | -1 (N/A) | false | false | 0 |
  | LANDLORD | PRO | Landlord Pro | 149900 | 1439000 | 10 | -1 (N/A) | true | false | 0 |
  | LANDLORD | BUSINESS | Landlord Business | 499900 | 4799000 | -1 | -1 (N/A) | true | true | 0 |

  Notes:
  - TENANT BUSINESS does not exist (only FREE and PRO for tenants)
  - Landlord FREE maxScoringViews = -1 because landlords don't have scoring view limits (they see candidate scores always, premium scoring is the gate)
  - Tenant FREE maxProperties = -1 because property limits are landlord-only
  - Annual prices = ~80% of monthly*12 (2 months free)
  - scoringViewPrice only applies to Tenant Free (micropayment per extra view)

  **Script structure:**
  ```typescript
  import { PrismaClient } from '@prisma/client';

  const prisma = new PrismaClient();

  async function main() {
    const plans = [
      { planType: 'TENANT', tier: 'FREE', name: 'Tenant Gratis', ... },
      // etc.
    ];

    for (const plan of plans) {
      await prisma.subscriptionPlanConfig.upsert({
        where: { planType_tier: { planType: plan.planType, tier: plan.tier } },
        update: { ...plan },  // Update prices if already seeded
        create: { ...plan },
      });
    }

    console.log('Seeded 5 subscription plan configs');
  }

  main()
    .catch(console.error)
    .finally(() => prisma.$disconnect());
  ```
