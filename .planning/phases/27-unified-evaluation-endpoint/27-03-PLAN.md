```typescript
@ApiTags('Evaluations')
@ApiBearerAuth()
@UseGuards(SupabaseAuthGuard, RolesGuard)
@Roles(Role.LANDLORD)  // AGENT inherits via RolesGuard
@Controller('evaluations')
export class EvaluationsController {
  constructor(private readonly evaluationsService: EvaluationsService) {}

  @Post(':applicationId')
  @HttpCode(202)
  @ApiOperation({ summary: 'Solicitar evaluación unificada del aplicante' })
  @ApiResponse({ status: 202, description: 'Evaluación iniciada' })
  @ApiResponse({ status: 402, description: 'Créditos insuficientes' })
  @ApiResponse({ status: 403, description: 'Sin suscripción activa o sin acceso' })
  @ApiResponse({ status: 429, description: 'Límite PRO de 30/mes alcanzado' })
  async requestEvaluation(
    @Param('applicationId', ParseUUIDPipe) applicationId: string,
    @CurrentUser() user: { id: string },
  ) {
    return this.evaluationsService.requestEvaluation(applicationId, user.id);
  }

  @Get(':applicationId/result')
  @ApiOperation({ summary: 'Consultar resultado de evaluación (polling)' })
  @ApiResponse({ status: 200, description: 'Estado actual de la evaluación' })
  @ApiResponse({ status: 404, description: 'No se encontró evaluación para esta solicitud' })
  async getResult(
    @Param('applicationId', ParseUUIDPipe) applicationId: string,
    @CurrentUser() user: { id: string },
  ) {
    return this.evaluationsService.getResult(applicationId, user.id);
  }
}
```

Use the same guard pattern as other controllers (check landlord.controller.ts or agent-credits.controller.ts for import paths). Use `@CurrentUser()` decorator from the auth module.

**1. `requestEvaluation(applicationId: string, requestedBy: string): Promise<{ runId: string; status: string }>`**

Flow (MUST follow this exact order):

a. **Load application + verify ownership:**
```typescript
const application = await this.prisma.application.findUniqueOrThrow({
  where: { id: applicationId },
  include: { property: true, documents: true },
});
if (application.property.landlordId !== requestedBy) {
  throw new ForbiddenException('No tenés acceso a esta solicitud');
}
```
Use `findUniqueOrThrow` — returns 404 automatically if application doesn't exist.

b. **Idempotency check:** Query `evaluationResult.findUnique({ where: { applicationId } })`. If exists and status is NOT `FAILED`, return `{ runId: existing.runId, status: existing.status }` — do NOT re-charge.

c. **If existing result is FAILED**, delete it first (so a new one can be created with the @unique constraint): `evaluationResult.delete({ where: { applicationId } })`.

d. **Validate plan + charge credits** — call `this.validateAndCharge(requestedBy)` (see method below).

e. **Build payload from application data:**
```typescript
const payload: TenantScoringPayload = {
  personalInfo: application.personalInfo as Record<string, unknown> | null,
  employmentInfo: application.employmentInfo as Record<string, unknown> | null,
  incomeInfo: application.incomeInfo as Record<string, unknown> | null,
  referencesInfo: application.referencesInfo as Record<string, unknown> | null,
  documents: application.documents.map((d) => ({
    type: d.type,
    storagePath: d.storagePath,
  })),
};
```

f. **Call micro:** `const { runId } = await this.agentMicroClient.startEvaluation(payload);`

g. **Persist in transaction:**
```typescript
await this.prisma.$transaction(async (tx) => {
  const evalResult = await tx.evaluationResult.create({
    data: { applicationId, requestedBy, runId, status: 'PENDING' },
  });
  await tx.evaluationTransaction.create({
    data: {
      evaluationId: evalResult.id,
      userId: requestedBy,
      tier: chargeResult.tier,
      amountPaidCop: chargeResult.amountPaidCop,
      creditsDeducted: chargeResult.creditsDeducted,
    },
  });
});
```

h. **Increment PRO monthly counter** (if tier is PRO): call `this.incrementEvaluationCount(requestedBy)`.

i. **Return** `{ runId, status: 'PENDING' }`.

**2. `getResult(applicationId: string, userId: string): Promise<EvaluationResponseDto>`**

a. Load `evaluationResult` by `applicationId`. If not found, throw `NotFoundException('No se encontró evaluación para esta solicitud')`.

b. Verify ownership: load application with property, check `application.property.landlordId === userId`. Throw 403 if not.

c. **If status is PENDING**, attempt poll:
```typescript
const microResult = await this.agentMicroClient.pollResult(evaluation.runId);
if (microResult) {
  // Update DB with result
  const updated = await this.prisma.evaluationResult.update({
    where: { id: evaluation.id },
    data: { status: 'COMPLETED', result: microResult },
  });
  return this.toResponseDto(updated);
}
```
If poll returns null, return current PENDING status as-is.

d. If status is COMPLETED or FAILED, return stored data directly.

**3. `validateAndCharge(userId: string): Promise<{ tier: string; amountPaidCop: number; creditsDeducted: number }>`** (private)

a. Get active subscription: `const subscription = await this.subscriptionsService.getActiveSubscription(userId)`. If null, throw `ForbiddenException('Se requiere una suscripción activa para solicitar evaluaciones')`.

b. Extract tier from subscription plan config. The subscription object has `plan` with `tier` property — check how `getActiveSubscription` returns it (look at SubscriptionsService).

c. **PRO monthly limit check:** If tier is PRO:
   - Call `this.getMonthlyEvaluationCount(userId)`
   - If count >= 30, throw `HttpException('Límite de 30 evaluaciones mensuales alcanzado', 429)`

d. **FLEX tier:** Return `{ tier: 'FLEX', amountPaidCop: 0, creditsDeducted: 0 }` — skip credit deduction entirely.

e. **STARTER/PRO tier:** Call `this.agentCreditsService.deductCredits(userId, 1, applicationId_not_needed_here)`. If deduction fails (returns false or throws), throw `BadRequestException('Saldo de créditos insuficiente. Comprá más créditos para continuar.')`.
   - Get `evaluationCreditPrice` from the plan config for the amountPaidCop field.
   - Return `{ tier, amountPaidCop: planConfig.evaluationCreditPrice, creditsDeducted: 1 }`.

IMPORTANT: Check `deductCredits` signature in AgentCreditsService — it takes `(userId, amount, applicationId?)`. Pass `undefined` for applicationId here since the deduction is for the evaluation, not a specific application context in the credits system.
