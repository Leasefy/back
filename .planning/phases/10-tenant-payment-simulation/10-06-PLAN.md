    ```typescript
    @Injectable()
    export class DisputesService {
      constructor(
        private readonly prisma: PrismaService,
        private readonly receiptStorage: ReceiptStorageService,
      ) {}

      /**
       * Create dispute for rejected payment request.
       * TPAY-11, TPAY-12
       */
      async create(
        paymentRequestId: string,
        tenantId: string,
        reason: string,
        evidenceFile?: Express.Multer.File,
      ): Promise<PaymentDispute> {
        // 1. Verify request is rejected and belongs to tenant
        const request = await this.prisma.tenantPaymentRequest.findFirst({
          where: {
            id: paymentRequestId,
            tenantId,
            status: 'REJECTED',
          },
          include: { lease: true },
        });

        if (!request) {
          throw new NotFoundException(
            'Rejected payment request not found. Only rejected requests can be disputed.',
          );
        }

        // 2. Check no existing dispute
        const existingDispute = await this.prisma.paymentDispute.findUnique({
          where: { paymentRequestId },
        });

        if (existingDispute) {
          throw new BadRequestException('Dispute already exists for this payment request');
        }

        // 3. Upload additional evidence if provided
        let evidencePath: string | undefined;
        if (evidenceFile) {
          // Reuse receipt storage with different prefix
          evidencePath = await this.receiptStorage.upload(
            request.leaseId,
            `dispute-${paymentRequestId}`,
            evidenceFile,
          );
        }

        // 4. Create dispute
        const dispute = await this.prisma.paymentDispute.create({
          data: {
            paymentRequestId,
            reason,
            additionalEvidencePath: evidencePath,
            status: 'OPEN',
          },
        });

        // 5. Update request status to DISPUTED
        await this.prisma.tenantPaymentRequest.update({
          where: { id: paymentRequestId },
          data: { status: 'DISPUTED' },
        });

        return dispute;
      }

      /**
       * Get disputes for a tenant.
       */
      async findByTenant(tenantId: string): Promise<PaymentDispute[]> {
        return this.prisma.paymentDispute.findMany({
          where: {
            paymentRequest: { tenantId },
          },
          include: {
            paymentRequest: {
              select: {
                amount: true,
                periodMonth: true,
                periodYear: true,
                rejectionReason: true,
                lease: {
                  select: { propertyAddress: true },
                },
              },
            },
          },
          orderBy: { createdAt: 'desc' },
        });
      }

      /**
       * Get single dispute with tenant access verification.
       */
      async findById(disputeId: string, tenantId: string): Promise<PaymentDispute> {
        const dispute = await this.prisma.paymentDispute.findUnique({
          where: { id: disputeId },
          include: {
            paymentRequest: {
              include: {
                lease: { select: { propertyAddress: true } },
              },
            },
          },
        });

        if (!dispute) {
          throw new NotFoundException('Dispute not found');
        }

        if (dispute.paymentRequest.tenantId !== tenantId) {
          throw new ForbiddenException('You do not have access to this dispute');
        }

        return dispute;
      }

      /**
       * Get signed URL for additional evidence.
       */
      async getEvidenceUrl(disputeId: string, tenantId: string): Promise<{ url: string; expiresAt: Date }> {
        const dispute = await this.findById(disputeId, tenantId);

        if (!dispute.additionalEvidencePath) {
          throw new NotFoundException('No additional evidence attached to this dispute');
        }

        return this.receiptStorage.getSignedUrl(dispute.additionalEvidencePath);
      }
    }
    ```

       ```typescript
       @ApiTags('Payment Disputes')
       @Controller()
       export class DisputesController {
         constructor(private readonly disputesService: DisputesService) {}

         // POST /payment-requests/:id/dispute (TPAY-11, TPAY-12)
         @Post('payment-requests/:id/dispute')
         @Roles(Role.TENANT, Role.BOTH)
         @UseInterceptors(FileInterceptor('evidence'))
         @ApiConsumes('multipart/form-data')
         @ApiOperation({ summary: 'Open dispute for rejected payment' })
         @ApiResponse({ status: 201, description: 'Dispute created' })
         async createDispute(
           @Param('id') paymentRequestId: string,
           @Body() dto: CreateDisputeDto,
           @UploadedFile() evidenceFile: Express.Multer.File | undefined,
           @CurrentUser() user,
         ) {
           const dispute = await this.disputesService.create(
             paymentRequestId,
             user.id,
             dto.reason,
             evidenceFile,
           );
           return {
             message: 'Dispute opened successfully. Our support team will review your case.',
             dispute,
           };
         }

         // GET /disputes
         @Get('disputes')
         @Roles(Role.TENANT, Role.BOTH)
         @ApiOperation({ summary: 'List my disputes' })
         async listDisputes(@CurrentUser() user) {
           return this.disputesService.findByTenant(user.id);
         }

         // GET /disputes/:id
         @Get('disputes/:id')
         @Roles(Role.TENANT, Role.BOTH)
         @ApiOperation({ summary: 'Get dispute details' })
         async getDispute(
           @Param('id') disputeId: string,
           @CurrentUser() user,
         ) {
           return this.disputesService.findById(disputeId, user.id);
         }

         // GET /disputes/:id/evidence-url
         @Get('disputes/:id/evidence-url')
         @Roles(Role.TENANT, Role.BOTH)
         @ApiOperation({ summary: 'Get signed URL for additional evidence' })
         async getEvidenceUrl(
           @Param('id') disputeId: string,
           @CurrentUser() user,
         ) {
           return this.disputesService.getEvidenceUrl(disputeId, user.id);
         }
       }
       ```
