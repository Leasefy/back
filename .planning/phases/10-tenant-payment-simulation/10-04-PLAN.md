    @Injectable()
    export class PseMockService {
      constructor(private readonly prisma: PrismaService) {}

      /**
       * Get list of banks available for PSE.
       */
      getAvailableBanks(): Array<{ code: string; name: string }> {
        return Object.values(ColombianBank).map(code => ({
          code,
          name: BANK_DISPLAY_NAMES[code],
        }));
      }

      /**
       * Process PSE mock payment.
       * Uses document number last digit for deterministic results:
       * - 0: FAILURE (Fondos insuficientes)
       * - 1: FAILURE (Transaccion rechazada por el banco)
       * - 9: PENDING (Pendiente de verificacion bancaria)
       * - Others: SUCCESS
       */
      processPayment(dto: PseMockRequestDto): Omit<PseMockResponseDto, 'paymentRequestId'> {
        const transactionId = `PSE-${Date.now()}-${Math.random().toString(36).substring(2, 8).toUpperCase()}`;
        const lastDigit = dto.documentNumber.slice(-1);
        const bankName = BANK_DISPLAY_NAMES[dto.bankCode];
        const timestamp = new Date();

        if (lastDigit === '0') {
          return {
            transactionId,
            status: 'FAILURE',
            message: 'Fondos insuficientes en la cuenta bancaria',
            bankName,
            timestamp,
          };
        }

        if (lastDigit === '1') {
          return {
            transactionId,
            status: 'FAILURE',
            message: 'Transaccion rechazada por el banco',
            bankName,
            timestamp,
          };
        }

        if (lastDigit === '9') {
          return {
            transactionId,
            status: 'PENDING',
            message: 'Transaccion pendiente de verificacion bancaria',
            bankName,
            timestamp,
          };
        }

        return {
          transactionId,
          status: 'SUCCESS',
          message: 'Pago procesado exitosamente',
          bankName,
          timestamp,
        };
      }
    }
    ```

       @ApiTags('PSE Mock')
       @Controller('pse-mock')
       export class PseMockController {
         constructor(
           private readonly pseMockService: PseMockService,
           private readonly tenantPaymentsService: TenantPaymentsService,
         ) {}

         // GET /pse-mock/banks (TPAY-07)
         @Get('banks')
         @Public()  // Allow unauthenticated access to bank list
         @ApiOperation({ summary: 'Get list of banks for PSE' })
         getBanks() {
           return this.pseMockService.getAvailableBanks();
         }

         // POST /pse-mock/process (TPAY-08)
         @Post('process')
         @Roles(Role.TENANT, Role.BOTH)
         @ApiOperation({ summary: 'Process PSE mock payment' })
         async processPayment(
           @Body() dto: PseMockRequestDto,
           @CurrentUser() user,
         ): Promise<PseMockResponseDto> {
           // 1. Process PSE mock
           const result = this.pseMockService.processPayment(dto);

           // 2. If SUCCESS, create payment request
           if (result.status === 'SUCCESS') {
             const paymentRequest = await this.tenantPaymentsService.createFromPse(
               user.id,
               dto.leaseId,
               {
                 amount: dto.amount,
                 periodMonth: dto.periodMonth,
                 periodYear: dto.periodYear,
                 pseTransactionId: result.transactionId,
                 pseBankCode: dto.bankCode,
               },
             );
             return { ...result, paymentRequestId: paymentRequest.id };
           }

           return result;
         }
       }
       ```
