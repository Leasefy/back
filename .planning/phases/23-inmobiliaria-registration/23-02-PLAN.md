# Plan 23-02: Sistema de Invitaciones por Token

## Goal

Implementar el sistema completo de invitaciones con tokens para que los miembros invitados a una agencia puedan aceptar o rechazar su invitación. Actualmente los miembros quedan perpetuamente en estado `INVITED` sin forma de activarse.

#### 1b. Nuevo método: `getInvitationByToken(token: string)`

```typescript
async getInvitationByToken(token: string) {
  const member = await this.prisma.agencyMember.findUnique({
    where: { invitationToken: token },
    include: { agency: true }
  });
  if (!member) throw new NotFoundException('Invitación no encontrada');
  if (member.invitationExpiresAt < new Date()) throw new BadRequestException('Invitación expirada');
  if (member.status !== 'INVITED') throw new BadRequestException('Invitación ya usada');
  return member;
}
```

#### 1c. Nuevo método: `acceptInvitation(token: string, userId: string)`

```typescript
async acceptInvitation(token: string, userId: string) {
  const member = await this.getInvitationByToken(token);

  // Verificar que el usuario no sea ya miembro de esta agencia
  const existingMember = await this.prisma.agencyMember.findUnique({
    where: { agencyId_userId: { agencyId: member.agencyId, userId } }
  });
  if (existingMember && existingMember.id !== member.id) {
    throw new ConflictException('Ya eres miembro de esta agencia');
  }

  return this.prisma.agencyMember.update({
    where: { id: member.id },
    data: {
      userId,
      status: 'ACTIVE',
      invitationToken: null,      // Consumir el token
      invitationExpiresAt: null,
    }
  });
}
```

#### 1d. Nuevo método: `declineInvitation(token: string)`

```typescript
async declineInvitation(token: string) {
  const member = await this.getInvitationByToken(token);
  return this.prisma.agencyMember.update({
    where: { id: member.id },
    data: {
      status: 'INACTIVE',
      invitationToken: null,
      invitationExpiresAt: null,
    }
  });
}
```

#### 1e. Nuevo método: `resendInvitation(memberId: string, agencyId: string)`

```typescript
async resendInvitation(memberId: string, agencyId: string) {
  const member = await this.prisma.agencyMember.findFirst({
    where: { id: memberId, agencyId, status: 'INVITED' }
  });
  if (!member) throw new NotFoundException('Miembro no encontrado o ya está activo');

  const invitationToken = crypto.randomUUID();
  const invitationExpiresAt = new Date(Date.now() + 7 * 24 * 60 * 60 * 1000);

  const updated = await this.prisma.agencyMember.update({
    where: { id: memberId },
    data: { invitationToken, invitationExpiresAt }
  });
  // Disparar evento de reenvío de email
  return updated;
}
```

// GET /inmobiliaria/agency/invitations/:token — PÚBLICO (sin auth guard)
@Get('invitations/:token')
@Public()  // Decorator que bypasa SupabaseAuthGuard
async getInvitationInfo(@Param('token') token: string) {
  const member = await this.agencyService.getInvitationByToken(token);
  return {
    agencyName: member.agency.name,
    agencyCity: member.agency.city,
    role: member.role,
    invitedEmail: member.invitedEmail || member.user?.email,
    expiresAt: member.invitationExpiresAt
  };
}

// POST /inmobiliaria/agency/invitations/:token/accept — REQUIERE AUTH
@Post('invitations/:token/accept')
async acceptInvitation(
  @Param('token') token: string,
  @CurrentUser() user: AuthUser
) {
  return this.agencyService.acceptInvitation(token, user.id);
}

// POST /inmobiliaria/agency/invitations/:token/decline
@Post('invitations/:token/decline')
@Public()  // Puede no estar logueado al declinar desde email
async declineInvitation(@Param('token') token: string) {
  return this.agencyService.declineInvitation(token);
}

// POST /inmobiliaria/agency/members/:memberId/resend-invitation
@Post('members/:memberId/resend-invitation')
@UseGuards(AgencyMemberGuard)
async resendInvitation(
  @Param('memberId') memberId: string,
  @Request() req: any  // req.agencyId del guard
) {
  // Solo ADMIN puede reenviar
  if (req.agencyMemberRole !== 'ADMIN') throw new ForbiddenException();
  return this.agencyService.resendInvitation(memberId, req.agencyId);
}
